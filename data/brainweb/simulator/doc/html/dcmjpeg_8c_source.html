101 {
102  unsigned char c;
103  //int testindex=0;
104  int n=0;
105  int NiDHT=0;
106  int indexY=0;

108

114

116  jpg->MaxHuffVal=0;
117

121  NiDHT+=jpg->RawDHT[n];
122  if (jpg->RawDHT[n]!=0)
123  jpg->MaxHuffSz=n;
124  }
125 
126  for(n=1;n<16;n++)
127  {
128  if(jpg->RawDHT[n]>0)
129  {
130  jpg->RawDHTstart[n]=indexY+1;
131  for (i=1;i<(jpg->RawDHT[n]+1);i++)
132  {
133  indexY+=1;
134  c=fgetc(jpg->infp);
135  jpg->lHuffTable[indexY].HufVal=c;
136  jpg->MaxHuffVal=c;
137  jpg->lHuffTable[indexY].HufSz=n;
138  }
139  }
140  }
141  k=1;
142  Code=0;
143 
144  Si=jpg->lHuffTable[k].HufSz;
145 
146  while(1)
147  {
148  if (k>=NiDHT) break;
149  while( Si==jpg->lHuffTable[k].HufSz)
150  {
151  jpg->lHuffTable[k].HufCode=Code;
152  Code+=1;
153  k+=1;
154  }
155  if (k<NiDHT)
156  {
157  while (jpg->lHuffTable[k].HufSz>Si)
158  {
159  Code=Code<<1;

161  }
162  }
163  }
164

186 static BOOL ClbJpegReadHeader(ClbJpeg *jpg)
187 {
188  unsigned char gr;
189  unsigned char el;
190  unsigned char l2;
191  unsigned char l1;
192  unsigned int sztag;
193  long ouca=0;
194  BOOL HeaderEnd=0;
195  BOOL isLossLess=0;
196 
197  int tp;
198 
199  gr=fgetc(jpg->infp); //FF
200  el=fgetc(jpg->infp); //D8
201 
202  while(!HeaderEnd)
203  {
204  gr=fgetc(jpg->infp);
205  if(gr!=0xFF) {
206  printf("gr!=0xFF (=%02x)\n",gr);
207  return 0;
208  }
209  el=fgetc(jpg->infp);
210 
211  if ( (el==0xFF) || (el==0x01) || (el==0xD8) ||(el==0xD9) ||( (el>=0xD0) && (el<=0xD7) ))
212  ;
213  else
214  {
215  l1=fgetc(jpg->infp);
216  l2=fgetc(jpg->infp);
217  sztag=(l1*256)+l2-2; //tag lengh
218  ouca=ftell(jpg->infp);
219 
220  if (el==0xC3)
221  {
222  jpg->lSof.precision=fgetc(jpg->infp);
223 
224  l1=fgetc(jpg->infp);
225  l2=fgetc(jpg->infp);
226  jpg->lSof.Himg=(l1*256)+l2;
227 
228  l1=fgetc(jpg->infp);
229  l2=fgetc(jpg->infp);
230  jpg->lSof.Wimg=(l1*256)+l2;
231 
232  jpg->lSof.NbComponent=fgetc(jpg->infp);
233 
234  jpg->lSof.SofTabPos=ftell(jpg->infp);
235 
236  isLossLess=TRUE;
237  }
238 
239  if (el==0xC4)
240  {
241  ClbJpegFillHuffTable(jpg);
242  }
243  if (el==0xDA)
244  {
245  jpg->lSos.CompCount=fgetc(jpg->infp);
246  for (tp=0;tp<jpg->lSos.CompCount;tp++)
247  {
248  jpg->lSos.CompId=fgetc(jpg->infp);
249  jpg->lSos.CompDc=fgetc(jpg->infp);
250  }
251  jpg->lSos.SpectralSelStart=fgetc(jpg->infp);
252  jpg->lSos.SpectralSelEnd=fgetc(jpg->infp);
253  jpg->lSos.SuccessiveAp=fgetc(jpg->infp);
254  jpg->lSos.Sospttrans=(jpg->lSos.SuccessiveAp & 16);
255  HeaderEnd=1;
256  }
257  if (el==0xDD)
258  {
259  l1=fgetc(jpg->infp);
260  l2=fgetc(jpg->infp);
261  jpg->RestartInterval=(l1*256)+l2;
262  }
263 
264  fseek(jpg->infp,(ouca+sztag),0);
265  }
266  }
267 
268  if (!isLossLess) {
269  printf("NOT isLossLess\n");
270  return 0;
271  }
272  return 1;
273 
274 }

276 static int ClbJpegReadBit(ClbJpeg *jpg)
277 {
278  int r=0;
279  unsigned char c;
280  if(jpg->PosCurBit>8) // need lire octet suivant
281  {
282  jpg->ValCurByte=fgetc(jpg->infp);
283  if (jpg->ValCurByte==0xFF)
284  {
285  c=fgetc(jpg->infp);// est 00 ou restart marker: a skiper
286  if (c!=0)
287  {
288  jpg->ValCurByte=fgetc(jpg->infp);
289  jpg->PosCurBit=1;
290  jpg->MarkerFound=1;
291  return 0;
292  }
293  }
294  jpg->PosCurBit=2;
295  return (jpg->ValCurByte>>7);
296  }
297  else
298  {
299  r=(1&(jpg->ValCurByte>>(8-jpg->PosCurBit)));
300  jpg->PosCurBit+=1;
301  return r;

328  {
329  for(iX=0;iX<jpg->lSof.Wimg;iX++) // lit premire ligne
330  {
331  lbInc+=1;
332  if (lbInc>1)
333  lPredicted= jpg->DataImg[lbInc-1];
334  jpg->DataImg[lbInc]=lPredicted+ClbJpegDecodeDiff(jpg);
335 
336  if ( jpg->DataImg[lbInc] > ((1<<(jpg->lSof.precision))-1) )
337  jpg->DataImg[lbInc]= jpg->DataImg[lbInc]&mask;

339  jpg->DataImg[lbInc]= jpg->DataImg[lbInc]&mask;
340  }

342  for (iY=1;iY<jpg->lSof.Himg;iY++) //lit la suite

344  lbInc+=1;

347  jpg->DataImg[lbInc]=lPredicted+ClbJpegDecodeDiff(jpg);
348

352  jpg->DataImg[lbInc]= jpg->DataImg[lbInc]&mask;

359  lPredicted=( jpg->DataImg[lbInc-1]+ jpg->DataImg[lbInc-jpg->lSof.Wimg])>>1;

361  lPredicted= jpg->DataImg[lbInc-1]; // se base%pixel juste avant

367  jpg->DataImg[lbInc]= jpg->DataImg[lbInc]&mask;

369  }

372  else // il y a un define interval

375  {
376  jpg->MarkerFound=0;

404 static int ClbJpegDecodeDiff(ClbJpeg *jpg)
405 {
406  int lInput;
407  int lInputBits;
408  int lHufVal;
409  int lDiff;
410  int lI;
411  int resultat;
412  lHufVal = 666;
413  lInput = 0;
414  lInputBits = 0;
415 
416 
417  while (1)
418  {
419  lInputBits+=1;
420  lInput=(lInput<<1)+ClbJpegReadBit(jpg);
421  if (jpg->RawDHT[lInputBits]!=0)
422  {
423  for(lI=jpg->RawDHTstart[lInputBits];lI<(jpg->RawDHTstart[lInputBits]+jpg->RawDHT[lInputBits]);lI++)
424  {
425  if (lInput==jpg->lHuffTable[lI].HufCode)
426  lHufVal=jpg->lHuffTable[lI].HufVal;
427  }
428  }
429  if (lInputBits>=jpg->MaxHuffSz)
430  lHufVal=jpg->MaxHuffVal;
431  if (lHufVal<255) break;
432  }
433  if (lHufVal==0) resultat= 0;
434 
435  if ( (lHufVal>0) && (lHufVal<16))
436  {
437  lDiff=0;
438  if( ClbJpegReadBit(jpg)==1)
439  {
440  for (lI=1;lI<lHufVal;lI++)
441  {
442  lDiff=(lDiff<<1)+ClbJpegReadBit(jpg);
443  }
444 
445  resultat= (lDiff+(1<<(lHufVal-1)));
446  }
447  else
448  {
449  for (lI=1;lI<lHufVal;lI++)
450  lDiff=(lDiff<<1)+1-ClbJpegReadBit(jpg);
451  resultat= -(lDiff+(1<<(lHufVal-1)));
452  }
453  }
454 
455  return resultat;
