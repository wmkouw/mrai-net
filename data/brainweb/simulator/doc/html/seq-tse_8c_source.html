69 void SeqTurboSpinEcho2D_0(SEQPARAM *seqparam,EXPERIENCE3D*expr)
70 {
71  int ntx,nty;
72  double fovx,fovy;
73  double te,tr,tacq,timp,acq_delay,tgp; /* ms */
74  double gx,gy,gz,gp;
75  int npz;
76  int tfact;
77  int i;
78  int Nenc,M,Nex,m,n;
79  int kmn1,kmn4,kmn5,k;
80 
81  EVENT* event, event_r;
82  event = &event_r;
83 
84  printf("Starting SeqTurboSpinEcho2D \n");
85  ntx = expr->ntx;
86  nty = expr->nty;
87  tacq = expr->readout_time*1.e+3;
88  fovx = expr->fovx;
89  fovy = expr->fovy;
90 
91  te = seqparam->te;
92  tr = seqparam->tr;
93  tfact = (int) seqparam->tfact;
94 
95  CheckTimesSE( tacq, te, tr);
96  timp=0.3; /*RF pulse duration in ms */
97  acq_delay = 0;
98  tgp = te/2.- tacq/2;
99  gx = (ntx-2.)/tacq/G_kHz_G/fovx;
100  gy = 2*M_PI*(nty-1)/G_rad_G/fovy/(tgp/1000);
101  gz = 0.0;
102  npz= 0;
103 
104  SetSpoilingFlag(event,ACTIVE); /* XY Spoiling after each line acquisition */
105  i=0;
106  DoWaiting(expr,event,ATT_INIT); /* To get an initial Magnetization M0 */
107  for(i=0;i<10;i++) /* Boucle de repetition pour un Mz saturee des la premiere acquisition*/
108  {
109  DoPulseRect(expr,event, 90.,timp);
110  DoWaiting(expr,event,te/2.0);
111  DoPulseRect(expr,event,180.,timp);
112  DoWaiting(expr,event,tr-te/2.0);
113  }
114 
115  Nenc = nty; /* Number of lines of the k space to acquire */
116  M=tfact; /* Number of echo per 90° RF pulse */
117  Nex = Nenc/M; /* Number of excitation to repeat */
118  printf("%d lines of the k space acquired in %d excitations of %d echos each \n",Nenc,Nex,M);
119 
120  for (m=1; m<= Nex; m++) /* Excitation m */
121  {
122  DoPulseRect(expr,event,90,timp);
123  DoGradient(expr,event,tacq/2,gx,0,0);
124  DoWaiting(expr,event,te/2-tacq/2);
125  for (n=1;n<=M;n++) /* Echo n */
126  {
127  DoPulseRect(expr,event,180.,timp);
128  /* kmn1 >> interleaving the strong magnitude echoes in the k space # k space subsampling, image periodisation */
129  kmn1=n-1+(m-1)*M-Nenc/2+Nenc/2;
130  /* kmn4 -> k space filling by a step of echo energy > a maximum of energy is contained in the HF half space */
131  kmn4=m-1+(n-1)*Nex-Nenc/2+Nenc/2;
132  /* kmn5 -> K space weighting by a step centered on the zero frequency */
133  if (m>Nex/2) kmn5= -(m-Nex/2+(n-1)*Nex/2)+Nenc/2;
134  else kmn5= (m-1+(n-1)*Nex/2)+Nenc/2;
135  k=kmn5; /* K space filling choice */
136  gp = (gy) * (k - Nenc / 2) / Nenc;
137  DoGradient(expr,event,tgp,0,-gp,0);
138  DoAcqFrequenceX(expr,event,gx,k,npz,acq_delay,MINUS);
139  DoGradient(expr,event,tgp,0,+gp,0);
140  i++;
141  }
142  DoWaiting(expr,event,tr-((tfact)*te)-te/2.0);
143  }
144  printf("Ending SeqTurboSpinEcho2D \n");
145 }
