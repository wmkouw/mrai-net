59 /* This half of the example shows how to read data from the JPEG decompressor.
60  * It's a bit more refined than the above, in that we show:
61  * (a) how to modify the JPEG library's standard error-reporting behavior;
62  * (b) how to allocate workspace using the library's memory manager.
63  *
64  * Just to make this example a little different from the first one, we'll
65  * assume that we do not intend to put the whole image into an in-memory
66  * buffer, but to send it line-by-line someplace else. We need a one-
67  * scanline-high JSAMPLE array as a work buffer, and we will let the JPEG
68  * memory manager allocate it for us. This approach is actually quite useful
69  * because we don't need to remember to deallocate the buffer separately: it
70  * will go away automatically when the JPEG object is cleaned up.
71  */

74 /*
75  * ERROR HANDLING:
76  *
77  * The JPEG library's standard error handler (jerror.c) is divided into
78  * several "methods" which you can override individually. This lets you
79  * adjust the behavior without duplicating a lot of code, which you might
80  * have to update with each future release.
81  *
82  * Our example here shows how to override the "error_exit" method so that
83  * control is returned to the library's caller when a fatal error occurs,
84  * rather than calling exit() as the standard error_exit method does.
85  *
86  * We use C's setjmp/longjmp facility to return control. This means that the
87  * routine which calls the JPEG library must first execute a setjmp() call to
88  * establish the return point. We want the replacement error_exit to do a
89  * longjmp(). But we need to make the setjmp buffer accessible to the
90  * error_exit routine. To do this, we make a private extension of the
91  * standard JPEG error handler object. (If we were using C++, we'd say we
92  * were making a subclass of the regular error handler.)
93  *
94  * Here's the extended error handler struct:
95  */
96 
97 struct my_error_mgr {
98  struct jpeg_error_mgr pub; /* "public" fields */
99  jmp_buf setjmp_buffer; /* for return to caller */
100 };
101 
102 typedef struct my_error_mgr * my_error_ptr;
103 
104 /*
105  * Here's the routine that will replace the standard error_exit method:
106  */
107 
108 METHODDEF(void)
109 my_error_exit (j_common_ptr cinfo)
110 {
111  /* cinfo->err really points to a my_error_mgr struct, so coerce pointer */
112  my_error_ptr myerr = (my_error_ptr) cinfo->err;
113 
114  /* Always display the message. */
115  /* We could postpone this until after returning, if we chose. */
116  (*cinfo->err->output_message) (cinfo);
117 
118  /* Return control to the setjmp point */
119  longjmp(myerr->setjmp_buffer, 1);
120 }
