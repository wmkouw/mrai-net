64 void TestFID1D(int objsize,int sigsize)
65  {
66  char fic[80];
67  int x,y,z;
68  double fovx,fovy,fovz;
69  double Tacq; /* in s */
70  int ro,t1,t2;
71  double acq_delay; /* in ms */
72  int tacq;
73  OBJECT3D *object;
74  EXPERIENCE3D* expr;
75  EVENT* event, event_r;
76  PPPVOLUME_COMPLEX_DOUBLE volrf;
77  PPPVOLUME_FLOAT volrecmod,volz,voly;
78 #ifdef HAVE_MPI
79  int rank;
80 #endif
81  fic[0]='\0';
82  printf("Start TestFID1D \n");
83  event = &event_r;
84  expr = AllocExperience();
85  x=objsize; y=1; z=1;
86  t1=150; t2=200; ro=10;
87  object=CreateHomogeneousObject(x,y,z,t1,t2,ro);
88  SetObjectSize(object,(float)0.2,(float)0.0002,(float)0.0002);
89  fovx=0.20; fovy=0.00020; fovz=0.00020;
90 
91  /* T2* weighting */
92  SetObjectDeltaB0(object,0.0000001);
93 
94  SetB0Experience(expr,1.5);
95  SetObjectExperienceComponent(expr,object,WATER,0);
96  SetFovExperience(expr,fovx,1,1,0,0,0);
97  tacq=800; /* in ms */
98  Tacq = tacq * 1.e-3; /* conversion in s */
99  SetAcqExperience(expr,sigsize,1,1,Tacq);
100  acq_delay=0;
101 
102  DoWaiting(expr,event,ATT_INIT);
103 
104  DoPulseRect(expr,event, 90., 300.);
105  DoAcqFrequenceX(expr,event,0,0,0,acq_delay,PLUS);
106 
107  /* Vizualisation of the T2* decay */
108  voly=GetSignalRFComponentFromExperience(expr,1);
109  DisplayVolX((PPPVOLUME)voly,0,0,"Antenne Y",0,"");
110  IdVolFree(voly);
111  /* Vizualisation of the T1 growing*/
112  volz=GetSignalRFComponentFromExperience(expr,2);
113 
114  event = &event_r;
115  expr=AllocExperience();
116 
117  DisplayVolX((PPPVOLUME)volz,0,0,"Antenne Z",0,"");
118  IdVolFree(volz);
119 
120  volrf = GetSignalRFComplexFromExperience(expr);
121  FreeExperience(expr);
122 
123  volrecmod = RecVolIFFTModuleFromVol(volrf);
124  volrf=NULL;
125 
126 #ifdef HAVE_MPI
127  MPI_Comm_rank(MPI_COMM_WORLD,&rank);
128  if (rank==0)
129  WriteVolRecUchar(volrecmod,"imafid1dsimri.acr");
130 #else
131  WriteVolRecUchar(volrecmod,"imafid1dsimri.acr");
132 #endif
133  IdVolFree(volrecmod);
134 
135  printf("End TestFID1D \n");
136  }
