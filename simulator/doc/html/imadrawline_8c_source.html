64 static int
65 CalculVoxelDroite(int premier, int ox, int oy, int fx, int fy, int *x, int *y)
66 {
67 
68  static int Dx; // Ecart en x entre debut et fin
69  static int Dy; // " y " " "
70  static short Incx; // Increment en x
71  static short Incy; // Increment en y
72  static int s;
73  static int Dx2; // Double de Dx
74  static int Dy2; // Double de Dy
75  static int Dxy; // Double de |Dx-Dy|
76  static int X; // coordonnee X courante
77  static int Y; // coordonnee Y courante
78  static int Voxel; // Compteur de voxel retourne
79 
80 #define SIGN(x) ((x)<0?-1:((x)>0?1:0))
81 
82  Voxel++;
83  /*
84  * Initialisation du trace de droite
85  */
86  if (premier) {
87  X = ox;
88  /*
89  * Sens de variation de la droite suivant X
90  */
91  Incx = SIGN(fx - X);
92  /*
93  * Variation totale suivant X
94  */
95  Dx = ABS(fx - X);
96  Dx2 = 2 * Dx;
97 
98  Y = oy;
99  /*
100  * Sens de variation de la droite
101  */
102  Incy = SIGN(fy - Y);
103  /*
104  * Variation totale suivant Y
105  */
106  Dy = ABS(fy - Y);
107  Dy2 = 2 * Dy;
108 
109  /*
110  * Servent a determiner si on va en diagonnale ou pas
111  */
112  Dxy = SIGN(Dx - Dy) * (Dy2 - Dx2);
113  if (Dx > Dy)
114  s = Dy2 - Dx;
115  else
116  s = Dx2 - Dy;
117 
118  Voxel = 0;
119  /*
120  * variation plus rapide suivant X
121  */
122  } else if (Dx > Dy) {
123  if (Voxel > Dx)
124  /*
125  * fin du trace
126  */
127  return 0;
128  if (s >= 0) {
129  Y += Incy;
130  s += Dxy;
131  } else {
132  s += Dy2;
133  }
134  X += Incx;
135  /*
136  * variation plus rapide suivant Y
137  */
138  } else if (Dx < Dy) {
139  if (Voxel > Dy)
140  /*
141  * fin du trace
142  */
143  return 0;
144  if (s >= 0) {
145  X += Incx;
146  s += Dxy;
147  } else {
148  s += Dx2;
149  }
150  Y += Incy;
151  /*
152  * variation égale suivant X et Y
153  */
154  } else {
155  if (Voxel > Dy)
156  /*
157  * fin du trace
158  */
159  return 0;
160  X += Incx;
161  Y += Incy;
162  }
163 
164  *x = X;
165  *y = Y;
166  return 1;
167 }

213 PPIMAGE_UCHAR
214 IdImaDrawLine(PPIMAGE_UCHAR ims, int value, int code,
215  int ox, int oy, int fx, int fy)
216 {
217  int x, y;
218  int init;
219  /*
220  * Controle de l'image *
221  */
222  if (!ims || IdImaType(ims) != IMA_UCHAR) {
223  IdErrno = IDERR_WRONG_TYPE;
224  return 0;
225  }
226 
227  if ((ox < 0) || (ox >= IdImaDimX(ims))
228  || (fx < 0) || (fx >= IdImaDimX(ims))
229  || (oy < 0) || (oy >= IdImaDimY(ims))
230  || (fy < 0) || (fy >= IdImaDimY(ims))
231  || (code > 3) || (code < 0)) {
232  IdErrno = IDERR_INVALID_NUMBER;
233  return 0;
234  }
235  /*
236  * Calcul du resultat ****
237  */
238  switch (code) {
239 
240  case 1:{
241  for (init =
242  CalculVoxelDroite(1, ox, oy, fx, fy, &x, &y);
243  init;
244  init =
245  CalculVoxelDroite(0, ox, oy, fx, fy, &x, &y)) {
246  ims[y][x] &= value;
247  }
248  }
249  break;
250 
251  case 2:{
252  for (init =
253  CalculVoxelDroite(1, ox, oy, fx, fy, &x, &y);
254  init;
255  init =
256  CalculVoxelDroite(0, ox, oy, fx, fy, &x, &y)) {
257  ims[y][x] |= value;
258  }
259  }
260  break;
261 
262  case 3:{
263  for (init =
264  CalculVoxelDroite(1, ox, oy, fx, fy, &x, &y);
265  init;
266  init =
267  CalculVoxelDroite(0, ox, oy, fx, fy, &x, &y)) {
268  ims[y][x] ^= value;
269  }
270  }
271  break;
272 
273  case 0:
274  default:{
275  for (init =
276  CalculVoxelDroite(1, ox, oy, fx, fy, &x, &y);
277  init;
278  init =
279  CalculVoxelDroite(0, ox, oy, fx, fy, &x, &y)) {
280  ims[y][x] = value;
281  }
282  }
283  }
284 
285  return ims;
286
